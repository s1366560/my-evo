# AI Agent-Driven Refactoring: Detecting Breaking Changes from "Smart" Renaming in Python

## 问题背景

AI 代码重构工具在进行"智能"重命名时（如函数、类、变量），容易引入破坏性变更（breaking changes）。本方案提供一套完整检测机制。

## 检测策略

### 1. AST 层面检测（抽象语法树）

```python
import ast
import difflib
from typing import Dict, List, Set, Tuple, Optional

class RenameDetector(ast.NodeVisitor):
    """检测重命名操作引入的破坏性变更"""
    
    def __init__(self):
        self.before_renames: Dict[str, ast.FunctionDef | ast.ClassDef] = {}
        self.after_renames: Dict[str, ast.FunctionDef | ast.ClassDef] = {}
        self.breaking_changes: List[dict] = []
        
    def extract_signature(self, node: ast.FunctionDef) -> Dict:
        """提取函数签名用于比较"""
        return {
            'name': node.name,
            'args': [arg.arg for arg in node.args.args],
            'defaults': [ast.unparse(d) if hasattr(ast, 'unparse') else '' 
                        for d in node.args.defaults],
            'returns': ast.unparse(node.returns) if node.returns else None,
            'decorators': [ast.unparse(d) for d in node.decorators]
        }
    
    def check_parameter_changes(self, before_fn, after_fn) -> List[str]:
        """检测参数变更"""
        changes = []
        
        # 检测参数数量变化
        before_args = len(before_fn['args'])
        after_args = len(after_fn['args'])
        
        if before_args != after_args:
            changes.append(f"Parameter count changed: {before_args} -> {after_args}")
        
        # 检测参数类型暗示变化（通过 docstring 分析）
        # 检测返回值类型变化
        before_returns = before_fn.get('returns')
        after_returns = after_fn.get('returns')
        
        if before_returns != after_returns:
            changes.append(f"Return type changed: {before_returns} -> {after_returns}")
            
        return changes
    
    def check_decorator_changes(self, before_fn, after_fn) -> List[str]:
        """检测装饰器变更"""
        changes = []
        before_decs = set(before_fn.get('decorators', []))
        after_decs = set(after_fn.get('decorators', []))
        
        removed = before_decs - after_decs
        added = after_decs - before_decs
        
        if removed:
            changes.append(f"Removed decorators: {removed}")
        if added:
            changes.append(f"Added decorators: {added}")
            
        return changes
```

### 2. 符号表级别检测

```python
import sys
import ast
from collections import defaultdict

class SymbolTableAnalyzer:
    """基于符号表的破坏性变更检测"""
    
    def __init__(self, source_code: str):
        self.source_code = source_code
        self.tree = ast.parse(source_code)
        self.symbols: Dict[str, List[ast.AST]] = defaultdict(list)
        self._build_symbol_table()
    
    def _build_symbol_table(self):
        """构建符号表"""
        for node in ast.walk(self.tree):
            if isinstance(node, (ast.FunctionDef, ast.ClassDef, ast.AsyncFunctionDef)):
                self.symbols[node.name].append(node)
            elif isinstance(node, ast.Name) and isinstance(node.ctx, ast.Store):
                self.symbols[f"var:{node.id}"].append(node)
    
    def detect_renamed_public_symbols(
        self, 
        before_symbols: Dict, 
        threshold: float = 0.8
    ) -> List[Tuple[str, str, float]]:
        """检测重命名的公共符号（基于相似度匹配）"""
        renamed = []
        
        before_public = {k: v for k, v in before_symbols.items() 
                        if not k.startswith('_') and not k.startswith('var:')}
        after_public = {k: v for k, v in self.symbols.items() 
                       if not k.startswith('_') and not k.startswith('var:')}
        
        before_names = set(before_public.keys())
        after_names = set(after_public.keys())
        
        # 找出新增和删除的
        removed = before_names - after_names
        added = after_names - before_names
        
        for rem in removed:
            # 尝试匹配相似名称
            for add in added:
                ratio = difflib.SequenceMatcher(None, rem, add).ratio()
                if ratio >= threshold:
                    renamed.append((rem, add, ratio))
                    
        return renamed
    
    def generate_breaking_report(
        self, 
        before_symbols: Dict,
        renamed: List[Tuple[str, str, float]]
    ) -> str:
        """生成破坏性变更报告"""
        report = ["# Breaking Changes Report", ""]
        
        if not renamed:
            report.append("✅ No breaking changes detected")
            return "\n".join(report)
            
        report.append(f"⚠️ Detected {len(renamed)} potential breaking renames:")
        for old_name, new_name, confidence in renamed:
            report.append(f"  - `{old_name}` → `{new_name}` (confidence: {confidence:.1%})")
            
        return "\n".join(report)
```

### 3. Type Hint 完整性检查

```python
import inspect
from typing import get_type_hints, get_origin, get_args

class TypeHintAnalyzer:
    """分析类型提示的兼容性"""
    
    @staticmethod
    def extract_function_signature(func) -> dict:
        """提取函数完整签名"""
        try:
            hints = get_type_hints(func)
        except Exception:
            hints = {}
            
        try:
            sig = inspect.signature(func)
        except (ValueError, TypeError):
            sig = None
            
        return {'hints': hints, 'signature': sig}
    
    @staticmethod
    def compare_signatures(before_func, after_func) -> List[str]:
        """比较两个函数的类型签名"""
        issues = []
        
        before_hints = TypeHintAnalyzer.extract_function_signature(before_func)
        after_hints = TypeHintAnalyzer.extract_function_signature(after_func)
        
        before_params = set(before_hints['hints'].keys())
        after_params = set(after_hints['hints'].keys())
        
        # 参数减少（破坏性）
        removed_params = before_params - after_params
        if removed_params:
            issues.append(f"Removed parameters (breaking): {removed_params}")
            
        # 新增必需参数（破坏性）
        before_defaults = len(before_hints['hints']) - len(before_params)
        new_required = after_params - before_params
        if new_required:
            issues.append(f"New required parameters (breaking): {new_required}")
            
        return issues
```

### 4. Import 依赖追踪

```python
class ImportTracker(ast.NodeVisitor):
    """追踪 import 变更"""
    
    def __init__(self):
        self.imports: Set[str] = set()
        self.from_imports: Dict[str, Set[str]] = defaultdict(set)
        
    def visit_Import(self, node: ast.Import):
        for alias in node.names:
            self.imports.add(alias.name)
            
    def visit_ImportFrom(self, node: ast.ImportFrom):
        if node.module:
            for alias in node.names:
                if alias.name == '*':
                    continue
                self.from_imports[node.module].add(alias.name)
    
    def detect_import_changes(self, before_tracker, after_tracker) -> dict:
        """检测 import 变更"""
        removed_imports = before_tracker.imports - after_tracker.imports
        added_imports = after_tracker.imports - before_tracker.imports
        
        # 检测 from X import Y 的变更
        removed_from = set(before_tracker.from_imports.keys()) - \
                       set(after_tracker.from_imports.keys())
        modified_from = {}
        for mod in set(before_tracker.from_imports.keys()) & \
                      set(after_tracker.from_imports.keys()):
            rem = before_tracker.from_imports[mod] - after_tracker.from_imports[mod]
            add = after_tracker.from_imports[mod] - before_tracker.from_imports[mod]
            if rem or add:
                modified_from[mod] = {'removed': list(rem), 'added': list(add)}
                
        return {
            'removed_imports': list(removed_imports),
            'added_imports': list(added_imports),
            'removed_modules': list(removed_from),
            'modified_from_imports': modified_from
        }
```

### 5. Pre-commit Hook 集成

```python
# .pre-commit-config.yaml
repos:
  - repo: local
    hooks:
      - id: breaking-change-detector
        name: Detect Breaking Refactoring Changes
        entry: python scripts/detect_breaking_changes.py
        language: system
        stages: [pre-commit, pre-push]
        pass_filenames: true
        
# scripts/detect_breaking_changes.py
#!/usr/bin/env python3
"""
Pre-commit hook for detecting breaking changes from refactoring.
Requires baseline JSON from previous commit state.
"""

import json
import sys
import subprocess
from pathlib import Path

def get_baseline_symbols(repo_path: str) -> dict:
    """获取基准版本的符号表"""
    # 从 git history 或缓存获取
    baseline_file = Path(repo_path) / '.refactor_baseline.json'
    if baseline_file.exists():
        return json.loads(baseline_file.read_text())
    return {}

def main():
    changed_files = sys.argv[1:] if len(sys.argv) > 1 else []
    
    for filepath in changed_files:
        if not filepath.endswith('.py'):
            continue
            
        # 运行 AST 分析
        result = subprocess.run(
            ['python', '-m', 'refactor_detector.analyze', filepath],
            capture_output=True,
            text=True
        )
        
        if result.returncode != 0:
            print(f"❌ Breaking changes detected in {filepath}")
            print(result.stdout)
            sys.exit(1)
            
    print("✅ No breaking changes detected")

if __name__ == '__main__':
    main()
```

### 6. Git Diff 级别集成

```python
import subprocess
from typing import Optional

class GitDiffAnalyzer:
    """基于 Git diff 的重命名检测"""
    
    @staticmethod
    def get_renamed_files(before_rev: str, after_rev: str) -> List[dict]:
        """获取重命名的文件"""
        result = subprocess.run(
            ['git', 'diff', '--name-status', '--diff-filter=R',
             before_rev, after_rev],
            capture_output=True,
            text=True
        )
        
        renamed = []
        for line in result.stdout.strip().split('\n'):
            if not line:
                continue
            parts = line.split('\t')
            if len(parts) >= 2:
                renamed.append({
                    'before': parts[1],
                    'after': parts[2] if len(parts) > 2 else parts[1]
                })
        return renamed
    
    @staticmethod
    def analyze_rename_semantic(before_file: str, after_file: str) -> dict:
        """分析重命名的语义变化"""
        result = subprocess.run(
            ['git', 'diff', '--no-ext-diff', before_file, after_file],
            capture_output=True,
            text=True
        )
        
        # 分析 diff 统计
        additions = result.stdout.count('+')
        deletions = result.stdout.count('-')
        
        return {
            'additions': additions,
            'deletions': deletions,
            'change_density': (additions + deletions) / max(len(result.stdout), 1)
        }
```

## 集成使用示例

```python
def full_refactoring_audit(before_code: str, after_code: str) -> dict:
    """完整的重构审计"""
    before_tree = ast.parse(before_code)
    after_tree = ast.parse(after_code)
    
    # 1. AST 级别检测
    before_tracker = RenameDetector()
    after_tracker = RenameDetector()
    before_tracker.visit(before_tree)
    after_tracker.visit(after_tree)
    
    # 2. 符号表比较
    symbol_analyzer = SymbolTableAnalyzer(after_code)
    renamed = symbol_analyzer.detect_renamed_public_symbols(
        before_tracker.symbols
    )
    
    # 3. Import 追踪
    before_imports = ImportTracker()
    after_imports = ImportTracker()
    before_imports.visit(before_tree)
    after_imports.visit(after_tree)
    import_changes = after_imports.detect_import_changes(
        before_imports, after_imports
    )
    
    # 4. 生成报告
    return {
        'renamed_symbols': renamed,
        'import_changes': import_changes,
        'breaking_score': len(renamed) * 0.4 + len(import_changes.get('removed_imports', [])) * 0.6,
        'is_breaking': len(renamed) > 0 or len(import_changes.get('removed_imports', [])) > 0
    }
```

## 总结

本方案提供多层检测：
1. **AST 层面**：检测函数签名、装饰器、类型提示变化
2. **符号表层面**：通过相似度匹配检测重命名
3. **Import 层面**：追踪依赖变更
4. **Git 层面**：通过 diff 分析文件级重命名
5. **Pre-commit Hook**：在提交前阻止破坏性变更

建议在 AI 重构工具中集成此检测框架，并在 CI/CD 流程中强制执行。
